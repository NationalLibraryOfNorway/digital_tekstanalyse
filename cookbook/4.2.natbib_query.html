2.3. Søk Nasjonalbibliografiens metadata med dhlab#

from dhlab.metadata.natbib import metadata_query, metadata_from_urn, pretty_print_marc21json

q = [
        ["245", "a", "kongen"]
        ]

metadata_query(q, limit = 3)

['URN:NBN:no-nb_digibok_2011011206017',
 'URN:NBN:no-nb_digibok_2013022006043',
 'URN:NBN:no-nb_digibok_2007080901030']

metadata_query tar en liste med tupler, der hver tuppel en MARC 21 delfelt

urns = metadata_query(
    [
        ("100", "a", "Ibsen")
        ],
    limit=2
                    )
urns

['URN:NBN:no-nb_digibok_2010021813003', 'URN:NBN:no-nb_digibok_2009042804050']
